    header: "Caused By:"

     - {{HYPOXEMIA}} (70% or more)
     - {{OPIATE_OVERDOSE}} (60% or more)
     - {{STASIS_BAG}}
     - {{MORBUSINE_POISONING}} (50% or more)
 
  - type: effects
    header: "Effects:"

     - {{ACIDOSIS}}
     - Inability to speak

    header: "Treatments:"
    order: 4

     - Being conscious
