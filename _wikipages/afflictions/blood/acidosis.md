  - type: caused_by
    header: "Caused By:"
    order: 2
    text: |
     - {{HYPOVENTILATION}}
     - {{RESPIRATORY_ARREST}}
     - {{CARDIAC_ARREST}}
     - {{KIDNEY_DAMAGE}} (80% or more)
     - {{BLOOD_PACKS}} who's donor had acidosis (acidosis gained will be the donor's acidosis level at time of extraction divided by 5)
     - {{SALINE}}

  - type: effects
    header: "Effects:"
    order: 3
    text: |
     - {{SEIZURE}} (at 60% or more)
     - {{COMA}} (at 60% or more)
     - {{INCREASED_HEARTRATE}} (at 20% or more)
     - {{FIBRILLATION}}
     - {{HEADACHE}} (at 20% or more)
     - {{CONFUSION}} (at 15% or more)
     - {{WEAKNESS}} (at 35% or more)

  - type: treatments
    header: "Treatments:"
    order: 4
    text: |
     - Treating the causes
     - Inducing {{ALKALOSIS}} (preferably with {{VOMITING}})
     - Transfusing neutral {{BLOOD_PACKS}} (unless the patient has 0% blood loss)
     - {{DIALYSIS_FILTER}}
